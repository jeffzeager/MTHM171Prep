\maketitle

\begin{problem}
Let $f(x) = \frac{x - 1}{x + 3}$.  Find the asymptote(s) of the graph of $y = f(x)$, if any exist.
\begin{selectAll}
\choice{$x = 1$}
\choice[correct]{$y = 1$}
\choice[correct]{$x = -3$}
\choice{$y = -3$}
\choice{None exist}
\end{selectAll}
\end{problem}
